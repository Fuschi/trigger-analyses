var_names <- colnames(trigger_hourly$data)
cor_order <- cor(trigger_hourly$data, method = "pearson",
                 use = "pairwise.complete.obs")
hc <- hclust(as.dist(1 - cor_order), method = "complete")
var_names <- var_names[hc$order]




corr_long <- expand_grid(var_y = var_names, var_x = var_names) %>%
  mutate(row_id = match(var_y, var_names),
         col_id = match(var_x, var_names),
         method = case_when(row_id < col_id ~ "spearman",
                            row_id > col_id ~ "pearson",
                            TRUE ~ NA_character_)) %>%
  filter(!is.na(method)) %>%
  mutate(correlation = pmap_dbl(
    .l = list(var_y, var_x, method),
    .f =\(y, x, m) cor(trigger_hourly$data[[y]],
                       trigger_hourly$data[[x]],
                       method = m, use = "pairwise.complete.obs"))) %>%
  mutate(var_x = factor(var_x, levels = var_names),
         var_y = factor(var_y, levels = rev(var_names))) 

rm(var_names, cor_order, hc)
p_cor <- corr_long %>%
  ggplot(aes(x = var_x, y = var_y, fill = correlation)) +
  geom_tile(color = "cornsilk3", linewidth = 0.3) +
  geom_text(
    aes(label = ifelse(abs(correlation) >= 0.3, round(correlation, 2), "")),
    size = 3) +
  scale_fill_gradient2(limits = c(-1, 1), midpoint = 0, name = "Correlation") +
  coord_fixed() +
  labs(title = "Correlation matrix",
       subtitle = "Upper triangle: Spearman | Lower triangle: Pearson",
       x = NULL, y = NULL) +
  theme_minimal(base_size = 11) +
  theme(
    axis.text.x = element_text(angle = 45, hjust = 1, vjust = 1),
    axis.text.y = element_text(size = 9),
    panel.grid = element_blank(),
    plot.title = element_text(face = "bold"),
    legend.position = "right")

gt_coef <- broom.mixed::tidy(model_heartrate_global_log, effects = "fixed") %>%
  filter(
    term != "(Intercept)",
    !stringr::str_detect(term, "^hour")) %>%
  mutate(
    abs_estimate = abs(estimate),
    p_value = gtsummary::style_pvalue(p.value, digits = 3),
    significance = case_when(
      p.value < 0.001 ~ "***",
      p.value < 0.01  ~ "**",
      p.value < 0.05  ~ "*",
      p.value < 0.1   ~ ".",
      TRUE ~ ""
    )
  ) %>%
  arrange(desc(abs_estimate)) %>%
  select(term, estimate, std.error, significance) %>%
  gt() %>%
  fmt_number(
    columns = c(estimate, std.error),
    decimals = 3
  ) %>%
  cols_label(
    term = "Predictor",
    estimate = html("&beta;"),
    std.error = "SE",
    significance = "Significance"
  ) %>%
  tab_header(
    title = "Fixed effects associated with heart rate",
    subtitle = "Continuous predictors were scaled before model fitting"
  ) %>%
  tab_source_note(
    source_note = "β values refer to one-standard-deviation increase in each continuous predictor."
  )
