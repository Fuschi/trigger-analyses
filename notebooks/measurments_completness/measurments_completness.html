df_quality <- trigger_hourly %>%
  filter(!is.na(userId), type %in% c("sum", "mean")) %>%
  filter(date >= as.Date("2025-03-01"), date <= Sys.Date()) %>%
  select(userId, date, hour, source, measurement, n_records_total, n_obs_valid) %>%
  distinct() %>%
  group_by(source, measurement) %>%
  summarise(
    quality_ratio = sum(n_obs_valid, na.rm = TRUE) / sum(n_records_total, na.rm = TRUE),
    total_obs_valid = sum(n_obs_valid, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  arrange(desc(quality_ratio)) %>%
  mutate(measurement = fct_reorder(measurement, quality_ratio)) %>%
  filter(measurement != "accuracy")

measurement_levels <- df_quality %>%
  arrange(quality_ratio) %>%
  pull(measurement)
p_quality <- df_quality %>%
  ggplot(aes(x = measurement, y = quality_ratio, fill = source)) +
  geom_col() + coord_flip() +
  scale_y_continuous(
    labels = scales::label_percent(accuracy = 1),
    limits = c(0, 1)) +
  labs(title = "Measurement completeness",
       subtitle = "Proportion of valid observations for each measurement",
       x = NULL, y = "Valid / total ratio") +
  theme_minimal(base_size = 14) +
  theme(
    plot.title = element_text(face = "bold", size = 16),
    axis.title = element_text(face = "bold"),
    panel.grid.minor = element_blank(),
    panel.grid.major.y = element_blank()
  )

upset_df <- trigger_hourly %>%
  filter(!is.na(userId), type %in% c("sum", "mean"),
         source != "gps") %>%
  filter(date >= as.Date("2025-03-01"), date <= Sys.Date(),
         n_obs_valid >= 10) %>%
  select(userId, date, hour, source, measurement, value) %>%
  distinct() %>%
  unite("sample_id", userId, date, hour, sep = "_", remove = FALSE)

measurement_sets <- upset_df %>%
  distinct(measurement) %>%
  pull(measurement)

upset_df <- upset_df %>%
  select(sample_id, measurement, value) %>%
  filter(sample_id != "61_2025-07-30_0") %>%
  mutate(present = !is.na(value)) %>%
  distinct(sample_id, measurement, present) %>%
  pivot_wider(
    names_from = measurement,
    values_from = present,
    values_fill = FALSE
  )
